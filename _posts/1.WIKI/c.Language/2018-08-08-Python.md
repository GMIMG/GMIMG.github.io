### epsilon_comparison

```python
#epsilon_comparison
from math import fabs
import sys
def is_equal(a,b, w=0):
    """
    is_equal(a,b, w=0)->bool
    w는 가중치입니다. 
    w를 0부터 늘려가며 상대 오차범위를 조정해주세요.
    """
    ep=sys.float_info.epsilon
    diff=fabs(a-b)

    return diff <= max(fabs(a),fabs(b))*ep*(2**w) #epsilon * 가중치
```

```python
class It:
    def __init__(self, *args):
        self.data=args
        self.idx=0

    def __iter__(self):
        return self

    def __next__(self):
        if self.idx >= len(self.data):
            raise StopIteration
        ret=self.data[self.idx]
        self.idx+=1
        return ret

li=[4,6,1,7]
it=It(*li)
for elem in it:
    print(elem)
```
